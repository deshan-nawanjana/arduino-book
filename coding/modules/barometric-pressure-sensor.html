PRESSURE SENSOR භාවිතය

By Deshan Nawanjana

මෙම සංවේදකය මගින් පරිසර උෂ්ණත්වය සහ පීඩනය ගණනය කිරීමේ හැකියාව පවතී. එමෙන්ම මෙම සංවදෙකය ක්‍රියාත්මක කිරීමට සහ ඉන් ප්‍රතිදාන ලබා ගැනීමට පහත දැක්වෙන Library ගොනුව ඔබගේ Arduino IDE මෘදුකාංගයට එක් කරගන්න. තවද මෙම ක්‍රමලේඛණයේ එන ALTITUDE යන්නට ඔබගේ ප්‍රදේශයේ මුහුදු මට්ටමේ සිට උස මීටරවලින් ලබා දෙන්න. එය මෙහිදී පීඩනය ගණනය කිරීමේදී උපයෝගී කරගනී.

Download SFE_BMP180 Library
REQUIREMENTS
1 x ARDUINO BOARD
1 x BAROMATRIC PRESSURE SENSOR
SETUP
BMP(VIN) = 3V3
BMP(GND) = GND
BMP(SDA) = PIN(A4)
BMP(SCL) = PIN(A5) 
#include <Wire.h>
#include <SFE_BMP180.h>

#define ALTITUDE 35.6
SFE_BMP180 pressure;

void setup() {
  Serial.begin(9600);
  if (pressure.begin()) {
    Serial.println("BMP180 connected!");
  } else {
    Serial.println("BMP180 not connected!");
  }
}

void loop() {
  char status;
  double T, P, p0;

  Serial.print("You provided altitude: ");
  Serial.print(ALTITUDE, 0);
  Serial.println(" meters");

  status = pressure.startTemperature();
  if (status != 0) {
    delay(status);

    status = pressure.getTemperature(T);
    if (status != 0) {
      Serial.print("Temp: ");
      Serial.print(T, 1);
      Serial.println(" deg C");

      status = pressure.startPressure(3);

      if (status != 0) {
        delay(status);

        status = pressure.getPressure(P, T);
        if (status != 0) {
          Serial.print("Pressure measurement: ");
          Serial.print(P);
          Serial.println(" hPa (Pressure measured using temperature)");

          p0 = pressure.sealevel(P, ALTITUDE);
          Serial.print("Relative (sea-level) pressure: ");
          Serial.print(p0);
          Serial.println("hPa");
          Serial.print("====================");
          Serial.println("");
        }
      }
    }
  }
  delay(1000);
}
